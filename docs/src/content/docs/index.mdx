## Quick Example

```nix
{
  services.ngrok = {
    enable = true;
    authToken = "your-ngrok-token";
    
    tunnels = {
      web = {
        protocol = "https";
        port = 80;
        hostname = "myapp.example.com";
        
        oauth = {
          provider = "google";
          allow_domains = [ "mycompany.com" ];
        };
        
        response_header_add = {
          "X-Frame-Options" = "DENY";
          "Strict-Transport-Security" = "max-age=31536000";
        };
        
        circuit_breaker = 0.1;
      };
    };
  };
}
```
